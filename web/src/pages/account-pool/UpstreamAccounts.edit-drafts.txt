describe("SharedUpstreamAccountDetailDrawer edit drafts", () => {
  function renderDetailDrawer(accountId: number, options?: { open?: boolean }) {
    host = document.createElement("div");
    document.body.appendChild(host);
    root = createRoot(host);
    rerenderDetailDrawer(accountId, options);
  }

  function rerenderDetailDrawer(accountId: number, options?: { open?: boolean }) {
    act(() => {
      root?.render(
        <I18nProvider>
          <SystemNotificationProvider>
            <MemoryRouter>
              <SharedUpstreamAccountDetailDrawer
                accountId={accountId}
                open={options?.open ?? true}
                onClose={() => undefined}
              />
            </MemoryRouter>
          </SystemNotificationProvider>
        </I18nProvider>,
      )
    })
  }

  function createApiKeyItem(id: number, displayName: string) {
    return {
      id,
      kind: "api_key_codex" as const,
      provider: "codex" as const,
      displayName,
      groupName: "prod",
      status: "active",
      displayStatus: "active",
      enabled: true,
      isMother: false,
      maskedApiKey: `sk-${id}••••`,
      primaryWindow: null,
      secondaryWindow: null,
      credits: null,
      localLimits: null,
      tags: [],
      effectiveRoutingRule: defaultEffectiveRoutingRule,
    }
  }

  function createApiKeyDetail(
    item: ReturnType<typeof createApiKeyItem>,
    overrides: Record<string, unknown> = {},
  ) {
    return {
      ...item,
      history: [],
      note: "Original note",
      upstreamBaseUrl: `https://proxy.example.com/${item.id}`,
      localLimits: {
        primaryLimit: 100,
        secondaryLimit: 1000,
        limitUnit: "requests",
      },
      recentActions: [],
      ...overrides,
    }
  }

  function createApiKeyHookState(options?: {
    items?: ReturnType<typeof createApiKeyItem>[]
    selectedId?: number
    detail?: Record<string, unknown> | null
    saveAccount?: ReturnType<typeof vi.fn>
    saveRouting?: ReturnType<typeof vi.fn>
    runSync?: ReturnType<typeof vi.fn>
    removeAccount?: ReturnType<typeof vi.fn>
  }) {
    const items = options?.items ?? [createApiKeyItem(8, "Gateway Key")]
    const selectedId = options?.selectedId ?? items[0]?.id ?? null
    const selectedSummary =
      items.find((item) => item.id === selectedId) ?? items[0] ?? null
    return {
      items,
      groups: [],
      forwardProxyNodes: [],
      hasUngroupedAccounts: false,
      writesEnabled: true,
      routing: {
        writesEnabled: true,
        apiKeyConfigured: true,
        maskedApiKey: "pool-live••••",
      },
      selectedId,
      selectedSummary,
      detail:
        options?.detail === undefined
          ? selectedSummary
            ? createApiKeyDetail(selectedSummary)
            : null
          : options.detail,
      isLoading: false,
      isDetailLoading: false,
      listError: null,
      detailError: null,
      error: null,
      missingDetailAccountId: null,
      selectAccount: vi.fn(),
      refresh: vi.fn(),
      loadDetail: vi.fn(),
      beginOauthLogin: vi.fn(),
      beginRelogin: vi.fn(),
      beginOauthMailboxSession: vi.fn(),
      beginOauthMailboxSessionForAddress: vi.fn(),
      getOauthMailboxStatuses: vi.fn(),
      removeOauthMailboxSession: vi.fn(),
      getLoginSession: vi.fn(),
      updateOauthLogin: vi.fn(),
      completeOauthLogin: vi.fn(),
      createApiKeyAccount: vi.fn(),
      runImportedOauthValidation: vi.fn(),
      startImportedOauthValidationJob: vi.fn(),
      stopImportedOauthValidationJob: vi.fn(),
      importOauthAccounts: vi.fn(),
      saveAccount: options?.saveAccount ?? vi.fn(),
      saveRouting: options?.saveRouting ?? vi.fn(),
      saveGroupNote: vi.fn(),
      runBulkAction: vi.fn(),
      startBulkSyncJob: vi.fn(),
      getBulkSyncJob: vi.fn(),
      stopBulkSyncJob: vi.fn(),
      runSync: options?.runSync ?? vi.fn(),
      removeAccount: options?.removeAccount ?? vi.fn(),
      total: items.length,
      page: 1,
      pageSize: 20,
      metrics: {
        total: items.length,
        oauth: 0,
        apiKey: items.length,
        attention: 0,
      },
    }
  }

  it("keeps pristine edit drafts in sync with same-account detail refreshes", async () => {
    const items = [createApiKeyItem(8, "Gateway Key")]
    let state = createApiKeyHookState({
      items,
      selectedId: 8,
      detail: createApiKeyDetail(items[0], {
        displayName: "Gateway Key",
        note: "Gateway note",
      }),
    })
    hookMocks.useUpstreamAccounts.mockImplementation(() => state)

    renderDetailDrawer(8)
    await flushAsync()
    clickTab(/Edit/i)
    await flushAsync()

    state = {
      ...state,
      detail: createApiKeyDetail(items[0], {
        displayName: "Gateway Key refreshed",
        note: "Gateway note refreshed",
      }),
    }
    rerenderDetailDrawer(8)
    await flushAsync()

    const displayNameInput = document.body.querySelector(
      'input[name="detailDisplayName"]',
    )
    const noteInput = document.body.querySelector('textarea[name="detailNote"]')
    expect((displayNameInput as HTMLInputElement).value).toBe(
      "Gateway Key refreshed",
    )
    expect((noteInput as HTMLTextAreaElement).value).toBe(
      "Gateway note refreshed",
    )
  })

  it("invalidates the reopened draft session before passive effects flush", async () => {
    const saveRequest = deferred<any>()
    const saveAccount = vi.fn().mockImplementation(() => saveRequest.promise)
    const savedDetail = createApiKeyDetail(createApiKeyItem(8, "Saved Gateway Key"), {
      note: "Saved note",
    })
    let state = createApiKeyHookState({
      items: [createApiKeyItem(8, "Gateway Key")],
      selectedId: 8,
      detail: createApiKeyDetail(createApiKeyItem(8, "Gateway Key"), {
        note: "Original note",
      }),
      saveAccount,
    })
    hookMocks.useUpstreamAccounts.mockImplementation(() => state)

    const drawerState = {
      accountId: 8,
      open: true,
    }

    function RoutedDetailDrawer() {
      return (
        <SharedUpstreamAccountDetailDrawer
          accountId={drawerState.accountId}
          open={drawerState.open}
          onClose={() => undefined}
        />
      )
    }

    const router = createMemoryRouter(
      [
        {
          path: "/",
          element: <RoutedDetailDrawer />,
        },
      ],
      {
        initialEntries: ["/?step=initial"],
      },
    )

    host = document.createElement("div")
    document.body.appendChild(host)
    root = createRoot(host)
    const renderWithRouter = () => {
      root?.render(
        <I18nProvider>
          <SystemNotificationProvider>
            <RouterProvider router={router} />
          </SystemNotificationProvider>
        </I18nProvider>,
      )
    }

    act(() => {
      renderWithRouter()
    })
    await flushAsync()
    clickTab(/Edit/i)
    await flushAsync()
    setInputValue('input[name="detailDisplayName"]', "Pending save draft")
    clickButton(/Save changes/i)
    await flushAsync()

    await act(async () => {
      drawerState.open = false
      renderWithRouter()
      await router.navigate("/?step=closed")
      drawerState.open = true
      renderWithRouter()
      await router.navigate("/?step=reopened")
      state = {
        ...state,
        detail: savedDetail,
      }
      renderWithRouter()
      saveRequest.resolve(savedDetail)
      await Promise.resolve()
    })

    await flushAsync()
    clickTab(/Edit/i)
    await flushAsync()

    const displayNameInput = document.body.querySelector(
      'input[name="detailDisplayName"]',
    )
    expect((displayNameInput as HTMLInputElement).value).toBe("Gateway Key")
  })
